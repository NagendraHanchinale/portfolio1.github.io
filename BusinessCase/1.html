Rider-Driver Matching

Problem

The problem involves efficiently assigning drivers to incoming rider requests in a real-time ride-hailing system. Each rider request includes key attributes such as location, time of request, and user-specific preferences. The goal is to determine the optimal driver for each request in a way that minimizes the rider’s waiting time and the distance a driver must travel to pick up the rider. Additionally, the system must ensure a fair distribution of rides among available drivers to prevent overworking or underutilizing any single driver. Finally, the solution must be scalable and capable of handling over 1,000 ride requests per second to support high-demand scenarios.

1. Greedy Matching: “First Come, First Served”

This approach assigns each rider to the nearest available driver based on ETA at the time of request.

Example 1 – Greedy Matching Flow

  • At 13:30:01, Rider 1 places a request.
  • System matches Rider 1 with Driver A (shortest ETA).
  • At 13:30:04, Rider 2 places a request.
  • Driver A is now busy, so Rider 2 gets Driver B.

❗ But what if:

  • Driver A would have had a better ETA for Rider 2?
  • Driver B could’ve served Rider 1 better?

Fig. 1: Rider-Driver assignment timeline (Greedy model).

🔧 Greedy Matching Code (C++)

#include <bits/stdc++.h>

struct Record {
    int order_id;
    int driver_id;
    long long order_timestamp;
    double eta;
};

std::vector> greedy_matching(std::vector& data) {
    std::sort(data.begin(), data.end(), [](const Record& a, const Record& b) {
        return a.order_timestamp < b.order_timestamp;
    });

    std::vector> matches;
    std::unordered_set assigned_drivers;
    std::vector orders;

    for (const auto& rec : data) {
        if (orders.empty() || orders.back() != rec.order_id) {
            orders.push_back(rec.order_id);
        }
    }

    for (int order_id : orders) {
        std::vector candidates;
        for (const auto& rec : data) {
            if (rec.order_id == order_id && assigned_drivers.find(rec.driver_id) == assigned_drivers.end()) {
                candidates.push_back(rec);
            }
        }

        if (candidates.empty()) continue;

        auto best_match = *std::min_element(candidates.begin(), candidates.end(),
            [](const Record& a, const Record& b) {
                return a.eta < b.eta;
            });

        matches.emplace_back(best_match.order_id, best_match.driver_id, best_match.eta);
        assigned_drivers.insert(best_match.driver_id);
    }

    return matches;
}

int main() {
    std::vector data = {
        {1, 101, 1000, 5.0},
        {1, 102, 1000, 3.0},
        {2, 101, 1003, 7.0},
        {2, 103, 1003, 4.0}
    };

    auto results = greedy_matching(data);

    for (const auto& match : results) {
        int order_id, driver_id;
        double eta;
        std::tie(order_id, driver_id, eta) = match;
        std::cout << "Order " << order_id << " matched with Driver " << driver_id << " ETA: " << eta << " seconds\n";
    }

    return 0;
}

2. Bulk Matching (Hungarian Algorithm)

Rather than assigning rides immediately, this strategy waits briefly to observe multiple requests and assigns drivers optimally.

Why use Bulk Matching?

  • Avoids suboptimal decisions made in isolation (as in greedy approaches).
  • By considering multiple requests together, the system can optimize assignments.
    • Example: Rider 1 and Rider 2 submit requests within 5 seconds.
    • By observing both, the system may assign drivers in a way that minimizes the total ETA.

Fig. 2: Effect of waiting a short interval to optimize total matching.

Graph Representation

The problem is modeled as a bipartite graph:

  • Set A: Riders
  • Set B: Drivers
  • Edge Weights: ETA between rider-driver pairs

Goal:

  • Find a minimum cost matching across all edges.
  • Solved using the Hungarian Algorithm (Kuhn-Munkres Algorithm).

Fig. 3: Orders-Drivers bipartite graph

📌 Bulk Matching Code: Hungarian Method (C++)

// hungarian.cpp
#include <bits/stdc++.h>
using namespace std;

// Hungarian Algorithm implementation (for minimum cost)
class Hungarian {
public:
    int n;
    vector> cost;
    vector u, v, p, way;

    Hungarian(int n) : n(n), cost(n, vector(n)), u(n+1), v(n+1), p(n+1), way(n+1) {}

    void setCost(int i, int j, int c) {
        cost[i][j] = c;
    }

    int assignment() {
        for (int i = 1; i <= n; i++) {
            p[0] = i;
            int j0 = 0;
            vector minv(n+1, INT_MAX);
            vector used(n+1, false);
            do {
                used[j0] = true;
                int i0 = p[j0], j1 = 0, delta = INT_MAX;
                for (int j = 1; j <= n; j++) {
                    if (!used[j]) {
                        int cur = cost[i0-1][j-1] - u[i0] - v[j];
                        if (cur < minv[j]) {
                            minv[j] = cur;
                            way[j] = j0;
                        }
                        if (minv[j] < delta) {
                            delta = minv[j];
                            j1 = j;
                        }
                    }
                }
                for (int j = 0; j <= n; j++) {
                    if (used[j]) {
                        u[p[j]] += delta;
                        v[j] -= delta;
                    } else {
                        minv[j] -= delta;
                    }
                }
                j0 = j1;
            } while (p[j0] != 0);

            do {
                int j1 = way[j0];
                p[j0] = p[j1];
                j0 = j1;
            } while (j0);
        }

        // p[j] = matched row for column j
        int result = 0;
        for (int j = 1; j <= n; j++) {
            int i = p[j];
            result += cost[i-1][j-1];
        }
        return result;
    }

    vector> getMatches() {
        vector> matches;
        for (int j = 1; j <= n; j++) {
            matches.emplace_back(p[j]-1, j-1); // (order_index, driver_index)
        }
        return matches;
    }
};

// Tuple representing a record: (bulk_id, order_id, driver_id, eta)
struct Record {
    int bulk_id;
    int order_id;
    int driver_id;
    int eta;
};

// Sample bulk matching function
void bulkMatching(vector &records) {
    // Group records by bulk_id
    unordered_map> bulks;
    for (auto &r : records) {
        bulks[r.bulk_id].push_back(r);
    }

    // Track assigned drivers globally
    unordered_map assignedDrivers;

    vector> bulk_matches; // (bulk_id, order_id, driver_id, eta)

    for (auto &[bulk_id, bulk_records] : bulks) {
        // Filter out drivers already assigned
        vector filtered;
        for (auto &r : bulk_records) {
            if (!assignedDrivers[r.driver_id]) filtered.push_back(r);
        }

        if (filtered.empty()) continue;

        // Get unique orders and drivers
        vector unique_orders, unique_drivers;
        unordered_map order_to_idx, driver_to_idx;

        for (auto &r : filtered) {
            if (order_to_idx.find(r.order_id) == order_to_idx.end()) {
                order_to_idx[r.order_id] = unique_orders.size();
                unique_orders.push_back(r.order_id);
            }
            if (driver_to_idx.find(r.driver_id) == driver_to_idx.end()) {
                driver_to_idx[r.driver_id] = unique_drivers.size();
                unique_drivers.push_back(r.driver_id);
            }
        }

        int n = max(unique_orders.size(), unique_drivers.size());
        Hungarian hungarian(n);

        // Initialize cost matrix with large values
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                hungarian.setCost(i, j, 1000000);

        // Fill cost matrix with eta values (convert to int, assume seconds)
        for (auto &r : filtered) {
            int i = order_to_idx[r.order_id];
            int j = driver_to_idx[r.driver_id];
            hungarian.setCost(i, j, r.eta);
        }

        hungarian.assignment();

        auto matches = hungarian.getMatches();

        for (auto &[order_idx, driver_idx] : matches) {
            if (order_idx < (int)unique_orders.size() && driver_idx < (int)unique_drivers.size()) {
                // Find eta for the matched pair
                int order_id = unique_orders[order_idx];
                int driver_id = unique_drivers[driver_idx];
                int eta = 1000000;

                for (auto &r : filtered) {
                    if (r.order_id == order_id && r.driver_id == driver_id) {
                        eta = r.eta;
                        break;
                    }
                }

                if (eta < 1000000) {
                    bulk_matches.emplace_back(bulk_id, order_id, driver_id, eta);
                    assignedDrivers[driver_id] = true;
                    cout << "Bulk " << bulk_id << ": Order " << order_id << " assigned to Driver " << driver_id << " with ETA " << eta << endl;
                }
            }
        }
    }

    // bulk_matches contains the final optimal assignments
}

int main() {
    // Sample input
    vector records = {
        {1, 101, 201, 300},
        {1, 101, 202, 180},
        {1, 102, 201, 420},
        {1, 102, 203, 240},
        {2, 103, 204, 200},
        {2, 104, 205, 150}
    };

    bulkMatching(records);

    return 0;
}
